# 用法`use`

 1、IPA -> ARPAbet
    
    f = self._ipa_convertor.convert('ˈɛniˌwʌn, -wən')
    self.assertEqual(f, 'EH1 N IY0 W AH2 N')
 2、ARPAbet -> KK音标
 
    f = self._arpabet_convertor.convert_to_american_phonetic_alphabet("W IY1 L K IY0 N S N")
    self.assertEqual(f, 'ˈwilkinsn')
